# Requirements for auto-sklearn environment
# ------------------------------------------
# Este archivo está pensado para crear un entorno separado compatible con auto-sklearn.
#
# IMPORTANTE: auto-sklearn requiere:
#   - Linux (no funciona en Windows nativo, usar WSL o Docker)
#   - Python 3.8 o 3.9 (no compatible con Python >= 3.10)
#   - scikit-learn < 0.25
#
# Instalación recomendada con Conda (más fiable):
#   conda create -n autosklearn-env python=3.9 -c conda-forge auto-sklearn flaml autogoal
#   conda activate autosklearn-env
#   pip install -r requirements-autosklearn.txt
#
# Instalación con pip (requiere dependencias del sistema):
#   sudo apt install -y build-essential swig libopenblas-dev python3.9-dev
#   python3.9 -m pip install -r requirements-autosklearn.txt
